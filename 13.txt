// 13.	Write a C/C++ program for inter-process communication using pipes.

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int main() {
    int fd[2];           
    pid_t pid;
    char write_msg[] = "Hello from parent to child via pipe!";
    char read_msg[100];

   
    if (pipe(fd) == -1) {
        perror("pipe");
        exit(1);
    }

    
    pid = fork();

    if (pid < 0) {
        perror("fork");
        exit(1);
    }

    if (pid > 0) {
       
        close(fd[0]);  
        printf("Parent: Writing message to pipe...\n");
        write(fd[1], write_msg, strlen(write_msg) + 1);
        close(fd[1]);  
    } else {
        
        close(fd[1]);  
        read(fd[0], read_msg, sizeof(read_msg));
        printf("Child: Received message: %s\n", read_msg);
        close(fd[0]);  
    }

    return 0;
}
