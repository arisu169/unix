// 8.	Implementation of Signals in UNIX.

#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>


void handle_signal(int sig) {
    if (sig == SIGINT) {
        printf("\nCaught SIGINT (Ctrl+C). Program will not terminate immediately.\n");
    }
    else if (sig == SIGTERM) {
        printf("\nCaught SIGTERM. Cleaning up and exiting.\n");
        exit(0);
    }
}

int main() {
    // Register signal handlers
    signal(SIGINT, handle_signal);   
    signal(SIGTERM, handle_signal);  

    printf("Program running. PID: %d\n", getpid());
    printf("Press Ctrl+C to test SIGINT or run 'kill -TERM %d' from another terminal.\n", getpid());

    // Infinite loop to keep program alive for testing
    while (1) {
        printf("Working...\n");
        sleep(2);
    }

    return 0;
}
